!pp(
!bash
~~~
__files="$(find . -maxdepth 1 -iname '*.md' -not -name 'Summary.md' -not -iname '\.*' | sort -n)"
if ! [ -z "${__files}" ]; then
    echo '# Class'
    echo "${__files}" | while read -r __line; do
        echo
        echo "${__line%.md}" | sed -e 's|^\./|## |' 
        echo
        cat "${__line}" | sed -e 's/^#/###/'
        echo
        echo '\newpage'
    done
fi
~~~
)
